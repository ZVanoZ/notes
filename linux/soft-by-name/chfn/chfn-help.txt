Использование: chfn [параметры] [ПОЛЬЗОВАТЕЛЬ]

Параметры:
  -f, --full-name ПОЛН_ИМЯ      смена полного имени пользователя
  -h, --home-phone ДОМ_ТЕЛЕФОН  смена домашнего тел. пользователя
  -o, --other ДРУГАЯ_ИНФ        смена другой инф. GECOS о пользователе
  -r, --room НОМЕР_КОМНАТЫ      смена номера комнаты пользователя
  -R, --root КАТ_CHROOT         каталог, в который выполняется chroot
  -u, --help                    показать данное сообщение и закончить работу
  -w, --work-phone РАБОЧИЙ_ТЕЛ  смена офисного номера тел. пользователя
