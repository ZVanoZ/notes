Использование:
  flatpak run [ПАРАМЕТР…] ПРИЛОЖЕНИЕ [АРГУМЕНТ…] - Запустить приложение

Параметры справки:
  -h, --help                                   Показать параметры справки
  --help-all                                   Показать все параметры справки

Параметры приложения:
  -u, --user                                   Работать с пользовательскими установками
  --system                                     Работать с установками для всех пользователей (по умолчанию)
  --installation=ИМЯ                           Работать с установками для всех пользователей (без умолчаний)
  --arch=АРХИТЕКТУРА                           Используемая архитектура
  --command=КОМАНДА                            Команда для запуска
  --cwd=КАТАЛОГ                                Каталог, в котором запустить команду
  --branch=ВЕТВЬ                               Ветвь для использования
  -d, --devel                                  Использовать среду исполнения для разработчиков
  --runtime=СРЕДА_ИСПОЛНЕНИЯ                   Используемая среда исполнения
  --runtime-version=ВЕРСИЯ                     Версия среды исполнения
  --log-session-bus                            Журналировать вызовы пользовательской шины
  --log-system-bus                             Журналировать вызовы системной шины
  --log-a11y-bus                               Журналировать вызовы шины универсального доступа
  --no-a11y-bus                                Не проксировать вызовы шины универсального доступа
  --a11y-bus                                   Проксировать вызовы шины универсального доступа (по-умолчанию, если не включена изоляция)
  --no-session-bus                             Не проксировать вызовы пользовательской шины
  --session-bus                                Проксировать вызовы пользовательской шины (по-умолчанию, если не включена изоляция)
  --no-documents-portal                        Не запускать порталы
  --file-forwarding                            Включить проброс файлов
  --commit                                     Запустить указанный коммит
  --runtime-commit                             Использовать указанный коммит среды исполнения
  --sandbox                                    Запустить полностью изолированным
  -p, --die-with-parent                        Убить процессы, когда родительский процесс умирает
  --parent-pid=PID                             Использовать ID_ПРОЦЕССА как номер процесса для совместного использования пространств имен
  --parent-expose-pids                         Сделать процессы видимыми в родительском пространстве имен
  --parent-share-pids                          Использовать общее с родителем пространство имён ID процессов
  --instance-id-fd                             Записать ID экземпляра в указанный файловый дескриптор
  --app-path=ПУТЬ                              Использовать ПУТЬ вместо /app приложения
  --usr-path=ПУТЬ                              Использовать ПУТЬ вместо /usr среды исполнения
  -v, --verbose                                Показать отладочную информацию, -vv для подробных деталей
  --ostree-verbose                             Показать отладочную информацию OSTree
  --share=ОБЩ_КАТАЛОГ                          Добавить общий каталог с хостом
  --unshare=ОБЩ_КАТАЛОГ                        Убрать общий каталог с хостом
  --socket=СОКЕТ                               Пробросить сокет приложению
  --nosocket=СОКЕТ                             Убрать проброс сокета приложению
  --device=УСТРОЙСТВО                          Пробросить устройство приложению
  --nodevice=УСТРОЙСТВО                        Убрать проброс устройства приложению
  --allow=ФУНКЦИОНАЛ                           Разрешить использовать этот функционал
  --disallow=ФУНКЦИОНАЛ                        Запретить использование этого функционала
  --filesystem=ФАЙЛ_СИСТЕМА[:ro]               Пробросить файловую систему приложению (:ro только для чтения)
  --nofilesystem=ФАЙЛ_СИСТЕМА                  Не пробрасывать файловую систему для приложения
  --env=ПЕРЕМЕННАЯ=ЗНАЧЕНИЕ                    Установить переменную окружения
  --env-fd=ДЕСКРИПТОР                          Читать переменные окружения в формате env -0 из ДЕСКРИПТОРа
  --unset-env=ПЕРЕМЕННАЯ                       Удалить переменную из окружения
  --own-name=ИМЯ_DBUS                          Разрешить приложению владеть именем на сессионной шине
  --talk-name=ИМЯ_DBUS                         Разрешить приложению обмениваться с именем используя сессионную шину
  --no-talk-name=ИМЯ_DBUS                      Запрещать приложению обмениваться с именем используя сессионную шину
  --system-own-name=ИМЯ_DBUS                   Разрешить приложению владеть именем используя системную шину
  --system-talk-name=ИМЯ_DBUS                  Разрешить приложению обмениваться с именем используя системную шину
  --system-no-talk-name=ИМЯ_DBUS               Запретить приложению обмениваться с именем используя системную шину
  --add-policy=ПОДСИСТЕМА.КЛЮЧ=ЗНАЧЕНИЕ        Добавить параметр обобщенной политики
  --remove-policy=ПОДСИСТЕМА.КЛЮЧ=ЗНАЧЕНИЕ     Удалить параметр обобщенной политики
  --persist=ИМЯ_ФАЙЛА                          Постоянный путь к домашнему каталогу
