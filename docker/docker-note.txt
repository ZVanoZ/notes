-- создает контейнер, но не запускает его.
$ docker create 
-- позволяет переименовать контейнер.
$ docker rename
-- создает и запускает контейнер за одну операцию.
$ docker run
-- удаляет контейнер.
$ docker rm 
-- обновляет ограничения ресурсов контейнера.
$ docker update
